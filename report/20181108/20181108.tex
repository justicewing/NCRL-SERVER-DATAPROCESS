\section{测试平台Bug的解决}
\subsection{Bug1：向上取整的错误}
\lstset{language=C++}
\begin{lstlisting}
mapped_len = (G + 1) / Q - 1;
改为
mapped_len = (G - 1) / Q + 1;
\end{lstlisting}
\subsection{Bug2：译码矩阵长度选择偏小}
\lstset{language=C++}
\begin{lstlisting}
col_hbg_d = K_b * 1024 / R + 2;
col_hbg_d = col_hbg_d >= (K_b + 4) ? col_hbg_d : (K_b + 4);
col_hbg_d = col_hbg_d <= col_hbg ? col_hbg_d : col_hbg;
改为
col_hbg_d = K_b * 1024 / R + 3;
col_hbg_d = col_hbg_d >= (K_b + 4) ? col_hbg_d : (K_b + 4);
col_hbg_d = col_hbg_d <= col_hbg ? col_hbg_d : col_hbg;
\end{lstlisting}
